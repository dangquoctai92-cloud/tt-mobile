Google Drive / Sheets
📋 Cách kết nối Google Sheets (3 bước)
Bước 1: Mở Google Sheet của mày → menu Extensions → Apps Script
Bước 2: Dán code Apps Script vào (xem hướng dẫn bên dưới) → Deploy → New deployment → Web app → Anyone → Deploy
Bước 3: Copy URL deploy dán vào ô bên dưới → Lưu & Kiểm tra

// =============================================
// SuaChuaPro - Apps Script (copy toàn bộ vào)
// =============================================
const SHEET_NAME = "Phiếu Sửa Chữa";
const FOLDER_NAME = "SuaChuaPro_Images";
const HEADERS = ["ID","Ngày nhận","Khách hàng","SĐT","Hãng","Model","IMEI","Mật khẩu","Lỗi báo","Tình trạng máy","Chẩn đoán","Phương pháp sửa","Tiền LK","Công thợ","Tổng tiền","Trạng thái","Ngày trả","Ghi chú","Ảnh trước","Ảnh sau","Ảnh tình trạng","Ảnh sửa chữa"];

// Lấy hoặc tạo folder ảnh trên Drive
function getImgFolder() {
  const folders = DriveApp.getFoldersByName(FOLDER_NAME);
  if (folders.hasNext()) return folders.next();
  const f = DriveApp.createFolder(FOLDER_NAME);
  f.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  return f;
}

// Upload 1 ảnh base64 lên Drive, trả về link xem
function uploadImage(base64Data, fileName) {
  try {
    const match = base64Data.match(/^data:([^;]+);base64,(.+)$/);
    if (!match) return "";
    const mimeType = match[1];
    const bytes = Utilities.base64Decode(match[2]);
    const blob = Utilities.newBlob(bytes, mimeType, fileName);
    const folder = getImgFolder();
    const file = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return "https://drive.google.com/uc?id=" + file.getId();
  } catch(e) { return ""; }
}

// Upload nhiều ảnh, trả về mảng URL
function uploadImages(imgArray, ticketId, slotName) {
  if (!imgArray || !imgArray.length) return [];
  return imgArray.map((img, i) => {
    if (!img.data) return img.url || "";          // đã là URL rồi, giữ nguyên
    const ext = (img.name || "img").split(".").pop() || "jpg";
    const fname = "ticket" + ticketId + "_" + slotName + "_" + i + "." + ext;
    return uploadImage(img.data, fname);
  }).filter(Boolean);
}

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let ws = ss.getSheetByName(SHEET_NAME);
  if (!ws) {
    ws = ss.insertSheet(SHEET_NAME);
    ws.appendRow(HEADERS);
  } else if (ws.getLastRow() === 0) {
    ws.appendRow(HEADERS);
  }
  return ws;
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);

    if (data.action === "upsert") {
      const r = data.row;
      const ws = getSheet();

      // Upload ảnh nếu có base64 mới
      const beforeUrls  = uploadImages(r.before_images_raw  || [], r.id, "before");
      const afterUrls   = uploadImages(r.after_images_raw   || [], r.id, "after");
      const condUrls    = uploadImages(r.condition_images_raw|| [], r.id, "cond");
      const repairUrls  = uploadImages(r.repair_images_raw  || [], r.id, "repair");

      // Gộp URL cũ (đã có) + URL mới upload
      const mergeUrls = (existing, newUrls) => {
        const all = (existing || []).concat(newUrls);
        return all.filter(Boolean).join("|");
      };

      const vals = [
        r.id, r.date_received, r.customer_name, r.phone,
        r.brand, r.model, r.imei, r.password,
        r.problem, r.machine_condition, r.diagnosis, r.repair_method,
        r.parts_cost, r.labor_cost, r.total,
        r.status, r.date_returned, r.notes,
        mergeUrls(r.before_images_urls,  beforeUrls),
        mergeUrls(r.after_images_urls,   afterUrls),
        mergeUrls(r.condition_images_urls,condUrls),
        mergeUrls(r.repair_images_urls,  repairUrls)
      ];

      const ids = ws.getRange(2,1,Math.max(ws.getLastRow()-1,1),1).getValues().flat();
      const idx = ids.indexOf(r.id);
      if (idx >= 0) ws.getRange(idx+2,1,1,vals.length).setValues([vals]);
      else ws.appendRow(vals);

      // Trả về URL ảnh để app cập nhật lại
      return out({ok:true, imageUrls:{
        before_images:  mergeUrls(r.before_images_urls,  beforeUrls),
        after_images:   mergeUrls(r.after_images_urls,   afterUrls),
        condition_images: mergeUrls(r.condition_images_urls, condUrls),
        repair_images:  mergeUrls(r.repair_images_urls,  repairUrls)
      }});
    }

    if (data.action === "delete") {
      const ws = getSheet();
      const ids = ws.getRange(2,1,Math.max(ws.getLastRow()-1,1),1).getValues().flat();
      const idx = ids.indexOf(data.id);
      if (idx >= 0) ws.deleteRow(idx+2);
      return ok("deleted");
    }

    if (data.action === "getAll") {
      const ws = getSheet();
      const rows = ws.getDataRange().getValues();
      return out({ok:true, rows});
    }

    if (data.action === "ping") return ok("pong");

  } catch(err) { return out({ok:false, error:err.toString()}); }
}

function doGet(e) {
  try {
    const ws = getSheet();
    const rows = ws.getDataRange().getValues();
    return out({ok:true, rows});
  } catch(err) { return out({ok:false, error:err.toString()}); }
}

function ok(msg) { return out({ok:true, msg}); }
function out(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
