const SHEET_NAME = "Phiếu Sửa Chữa";
const USER_SHEET_NAME = "Tài Khoản";
const FOLDER_NAME = "SuaChuaPro_Images";
const HEADERS = ["ID","Ngày nhận","Khách hàng","SĐT","Hãng","Model","IMEI","Mật khẩu","Lỗi báo","Tình trạng máy","Chẩn đoán","Phương pháp sửa","Tiền LK","Công thợ","Tổng tiền","Trạng thái","Ngày trả","Ghi chú","Ảnh trước","Ảnh sau","Ảnh tình trạng","Ảnh sửa chữa"];
const USER_HEADERS = ["username","password","displayName","role"];
const HISTORY_SHEET_NAME = "TicketEditHistory";
const HISTORY_HEADERS = ["ts","ticket_id","action","by","note"];

function getChatRevision() {
  const p = PropertiesService.getScriptProperties();
  const v = p.getProperty('CHAT_REV');
  return v ? parseInt(v, 10) : 0;
}
function bumpChatRevision() {
  const n = getChatRevision() + 1;
  PropertiesService.getScriptProperties().setProperty('CHAT_REV', String(n));
  return n;
}

function findSheetByNameNormalized(ss, sheetName) {
  var sheets = ss.getSheets();
  var target = sheetName.normalize("NFC");
  for (var i = 0; i < sheets.length; i++) {
    var name = sheets[i].getName();
    if (name === sheetName || name.normalize("NFC") === target || name.normalize("NFD") === target) {
      return sheets[i];
    }
  }
  return null;
}

function getUserSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let ws = findSheetByNameNormalized(ss, USER_SHEET_NAME);
  if (!ws) {
    ws = ss.insertSheet(USER_SHEET_NAME);
    ws.appendRow(USER_HEADERS);
    ws.appendRow(["admin","admin123","Admin","admin"]);
  }
  return ws;
}

// Lấy hoặc tạo folder ảnh trên Drive
function getImgFolder() {
  const folders = DriveApp.getFoldersByName(FOLDER_NAME);
  if (folders.hasNext()) return folders.next();
  const f = DriveApp.createFolder(FOLDER_NAME);
  f.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  return f;
}

// Upload 1 ảnh base64 lên Drive, trả về link xem
function uploadImage(base64Data, fileName) {
  try {
    const match = base64Data.match(/^data:([^;]+);base64,(.+)$/);
    if (!match) return "";
    const mimeType = match[1];
    const bytes = Utilities.base64Decode(match[2]);
    const blob = Utilities.newBlob(bytes, mimeType, fileName);
    const folder = getImgFolder();
    const file = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return "https://drive.google.com/uc?id=" + file.getId();
  } catch(e) { return ""; }
}

// Upload nhiều ảnh, trả về mảng URL
function uploadImages(imgArray, ticketId, slotName) {
  if (!imgArray || !imgArray.length) return [];
  return imgArray.map((img, i) => {
    if (!img.data) return img.url || "";          // đã là URL rồi, giữ nguyên
    const ext = (img.name || "img").split(".").pop() || "jpg";
    const fname = "ticket" + ticketId + "_" + slotName + "_" + i + "." + ext;
    return uploadImage(img.data, fname);
  }).filter(Boolean);
}

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let ws = findSheetByNameNormalized(ss, SHEET_NAME);
  if (!ws) {
    ws = ss.insertSheet(SHEET_NAME);
    ws.appendRow(HEADERS);
  } else if (ws.getLastRow() === 0) {
    ws.appendRow(HEADERS);
  }
  return ws;
}

function getHistorySheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let ws = findSheetByNameNormalized(ss, HISTORY_SHEET_NAME);
  if (!ws) {
    ws = ss.insertSheet(HISTORY_SHEET_NAME);
    ws.appendRow(HISTORY_HEADERS);
  } else if (ws.getLastRow() === 0) {
    ws.appendRow(HISTORY_HEADERS);
  }
  return ws;
}

  try {
    const data = JSON.parse(e.postData.contents);

    if (data.action === "upsert") {
      const r = data.row;
      const ws = getSheet();

        return all.filter(Boolean).join("|");
      };

      const txt = (v) => v == null ? "" : String(v).trim();
      const normDate = (v) => {
        const s = txt(v);
        if (!s) return "";
        if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s;
        const m = s.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/);
        if (m) return m[3] + "-" + ("0"+m[2]).slice(-2) + "-" + ("0"+m[1]).slice(-2);
        return s;
      };
      const vals = [
        r.id, r.date_received, r.customer_name, txt(r.phone),
        r.brand, r.model, txt(r.imei), txt(r.password),
        r.problem, r.machine_condition, r.diagnosis, r.repair_method,
        r.parts_cost, r.labor_cost, r.total,
        r.status, normDate(r.date_returned), r.notes,
        mergeUrls(r.before_images_urls,  beforeUrls),
        mergeUrls(r.after_images_urls,   afterUrls),
        mergeUrls(r.condition_images_urls,condUrls),
        mergeUrls(r.repair_images_urls,  repairUrls)
      ];

      const ids = ws.getRange(2,1,Math.max(ws.getLastRow()-1,1),1).getValues().flat();
      const idx = ids.indexOf(r.id);
      const actionType = idx >= 0 ? "update" : "create";
      if (idx >= 0) ws.getRange(idx+2,1,1,vals.length).setValues([vals]);
      else ws.appendRow(vals);
      // Ép định dạng text cho SĐT / IMEI / Ngày trả để tránh Sheets tự đổi kiểu (mất số 0 đầu, scientific notation...)
      try {
        const rowNo = idx >= 0 ? (idx + 2) : ws.getLastRow();
        ws.getRange(rowNo, 4, 1, 1).setNumberFormat("@");   // SĐT
        ws.getRange(rowNo, 7, 1, 1).setNumberFormat("@");   // IMEI
        ws.getRange(rowNo, 17, 1, 1).setNumberFormat("@");  // Ngày trả
      } catch (eFmt) {}

      // Lưu nhật ký chỉnh sửa phiếu (admin dùng để xem lịch sử)
      try {
        const actor = r.updated_by || r.created_by || "";
        const summary = (r.customer_name || "") + " • " + (r.status || "");
        getHistorySheet().appendRow([Date.now(), r.id, actionType, actor, summary]);
      } catch(err) {}

      // Trả về URL ảnh để app cập nhật lại
      return out({ok:true, imageUrls:{
        before_images:  mergeUrls(r.before_images_urls,  beforeUrls),
        after_images:   mergeUrls(r.after_images_urls,   afterUrls),
        condition_images: mergeUrls(r.condition_images_urls, condUrls),
        repair_images:  mergeUrls(r.repair_images_urls,  repairUrls)
      }});
    }

    if (data.action === "delete") {
      const ws = getSheet();
      const ids = ws.getRange(2,1,Math.max(ws.getLastRow()-1,1),1).getValues().flat();
      const idx = ids.indexOf(data.id);
      if (idx >= 0) ws.deleteRow(idx+2);
      // Nhật ký xóa phiếu
      try {
        const actor = data.requestUser || data.updated_by || "";
        getHistorySheet().appendRow([Date.now(), data.id, "delete", actor, "Xóa phiếu"]);
      } catch(err) {}
      return ok("deleted");
    }

    if (data.action === "ticket_history_get") {
      const hs = getHistorySheet();
      const rows = hs.getDataRange().getValues();
      return out({ok:true, rows});
    }

    if (data.action === "ticket_history_clear") {
      const req = data.requestUser;
      if (!req) return out({ok:false, error:"no user"});
      const uws = getUserSheet();
      const urows = uws.getDataRange().getValues();
      const okAdmin = urows.slice(1).some(r => r[0] === req && String(r[3]).toLowerCase() === "admin");
      if (!okAdmin) return out({ok:false, error:"forbidden"});
      const hs = getHistorySheet();
      if (hs.getLastRow() > 1) hs.deleteRows(2, hs.getLastRow() - 1);
      return out({ok:true, msg:"cleared"});
    }

    if (data.action === "drive_storage") {
      // Quota tài khoản: Drive API about + bổ sung DriveApp (Workspace hay trả about rỗng). Thư mục app: đo lại qua API nếu getSize() = 0.
      function qNum(v) {
        if (v == null || v === "") return 0;
        var n = Number(String(v).replace(/,/g, ""));
        return isFinite(n) ? n : 0;
      }
      function scanFolder(folder) {
        let total = 0;
        let count = 0;
        const files = folder.getFiles();
        while (files.hasNext()) {
          const f = files.next();
          total += f.getSize();
          count++;
        }
        const subs = folder.getFolders();
        while (subs.hasNext()) {
          const sf = subs.next();
          const sub = scanFolder(sf);
          total += sub.total;
          count += sub.count;
        }
        return { total, count };
      }
      function sumFolderBytesDriveApi(folderId, token) {
        function walk(fid, depth) {
          if (depth > 30) return { b: 0, c: 0 };
          var bytes = 0, count = 0, pageToken = "";
          do {
            var url = "https://www.googleapis.com/drive/v3/files?pageSize=1000&fields=nextPageToken,files(id,mimeType,size)&q=" +
              encodeURIComponent("'" + fid + "' in parents and trashed = false");
            if (pageToken) url += "&pageToken=" + encodeURIComponent(pageToken);
            var r = UrlFetchApp.fetch(url, {
              headers: { Authorization: "Bearer " + token },
              muteHttpExceptions: true
            });
            if (r.getResponseCode() !== 200) return { b: bytes, c: count };
            var data = JSON.parse(r.getContentText());
            var files = data.files || [];
            for (var i = 0; i < files.length; i++) {
              var f = files[i];
              var mt = f.mimeType || "";
              if (mt === "application/vnd.google-apps.folder") {
                var sub = walk(f.id, depth + 1);
                bytes += sub.b;
                count += sub.c;
              } else {
                count++;
                if (f.size != null && f.size !== "") bytes += qNum(f.size);
              }
            }
            pageToken = data.nextPageToken || "";
          } while (pageToken);
          return { b: bytes, c: count };
        }
        try { return walk(folderId, 0); } catch (eW) { return { b: 0, c: 0 }; }
      }

      var driveAppUsed = 0;
      var driveAppLimit = 0;
      try {
        driveAppUsed = DriveApp.getStorageUsed();
        driveAppLimit = DriveApp.getStorageLimit();
      } catch (eDA) {}

      const folder = getImgFolder();
      const res = scanFolder(folder);
      var folderBytes = res.total;
      if (res.count > 0 && folderBytes === 0) {
        try {
          var tokFix = ScriptApp.getOAuthToken();
          var apiSum = sumFolderBytesDriveApi(folder.getId(), tokFix);
          if (apiSum.b > 0) folderBytes = apiSum.b;
        } catch (eFx) {}
      }

      var usageTotal = 0;
      var limitTotal = 0;
      var usageDrive = 0;
      var usagePhotos = 0;
      var usageMailOther = 0;
      var quotaSource = "fallback";

      var userEmail = "";
      try {
        var aboutUrl = "https://www.googleapis.com/drive/v3/about?fields=storageQuota,user";
        var tok = ScriptApp.getOAuthToken();
        var resp = UrlFetchApp.fetch(aboutUrl, {
          method: "get",
          headers: { Authorization: "Bearer " + tok },
          muteHttpExceptions: true
        });
        if (resp.getResponseCode() === 200) {
          var aj = JSON.parse(resp.getContentText());
          var sq = aj.storageQuota || {};
          usageTotal = qNum(sq.usage);
          limitTotal = qNum(sq.limit);
          var uDrive = qNum(sq.usageInDrive);
          var uTrash = qNum(sq.usageInDriveTrash);
          usageDrive = uDrive + uTrash;
          usagePhotos = qNum(sq.usageInPhotos);
          usageMailOther = Math.max(0, usageTotal - usageDrive - usagePhotos);
          quotaSource = "about_v3";
          if (aj.user && aj.user.emailAddress) {
            userEmail = aj.user.emailAddress;
          }
        }
      } catch (eA) {}

      if (quotaSource !== "about_v3") {
        if (driveAppUsed > 0) usageTotal = driveAppUsed;
        if (driveAppLimit > 0) limitTotal = driveAppLimit;
        usageDrive = usageTotal;
        usagePhotos = 0;
        usageMailOther = 0;
        quotaSource = "driveapp";
      } else {
        usageMailOther = Math.max(0, usageTotal - usageDrive - usagePhotos);
        if (usageTotal === 0 && driveAppUsed > 0) {
          usageTotal = driveAppUsed;
          usageDrive = driveAppUsed;
          usagePhotos = 0;
          usageMailOther = 0;
          quotaSource = "mixed";
        } else if (usageTotal > 0 && usageDrive === 0 && usagePhotos === 0 && usageMailOther === 0) {
          usageDrive = usageTotal;
          usageMailOther = 0;
        }
        if (limitTotal === 0 && driveAppLimit > 0) limitTotal = driveAppLimit;
      }

      if (!usageTotal && folderBytes > 0) usageTotal = folderBytes;
      if (!usageDrive && usageTotal > 0 && quotaSource === "mixed") usageDrive = usageTotal;

      var usedBytesFinal = usageTotal;
      var limitBytesFinal = limitTotal;
      var pctFinal = limitBytesFinal > 0 ? (usedBytesFinal / limitBytesFinal * 100) : 0;

      if (!userEmail) {
        try { userEmail = Session.getEffectiveUser().getEmail(); } catch(e) {}
      }
      if (!userEmail) {
        try { userEmail = Session.getActiveUser().getEmail(); } catch(e) {}
      }

      return out({
        ok: true,
        usedBytes: usedBytesFinal,
        limitBytes: limitBytesFinal,
        pct: pctFinal,
        usageInDrive: usageDrive,
        usageInPhotos: usagePhotos,
        usageGmailOther: usageMailOther,
        quotaSource: quotaSource,
        folderUsedBytes: folderBytes,
        fileCount: res.count,
        scannedAt: Date.now(),
        userEmail: userEmail
      });
    }

    if (data.action === "getAll") {
      const ws = getSheet();
      const rows = ws.getDataRange().getValues();
      return out({ok:true, rows});
    }

    if (data.action === "ping") return ok("pong");

    if (data.action === "chat_send") {
      const ws = getChatSheet();
      pruneChatSheetOlderThanDays(ws);
      const m = data.msg;
      const cellText = chatCellTextFromMsg(m);
      ws.appendRow([m.id, m.username, m.displayName, cellText, m.ts]);
      return out({ ok: true, msg: "sent", savedText: cellText });
    }

    if (data.action === "chat_clear") {
      const req = data.requestUser;
      if (!req) return out({ok:false, error:"no user"});
      const uws = getUserSheet();
      const urows = uws.getDataRange().getValues();
      const okAdmin = urows.slice(1).some(r => r[0] === req && String(r[3]).toLowerCase() === "admin");
      if (!okAdmin) return out({ok:false, error:"forbidden"});
      const cws = getChatSheet();
      if (cws.getLastRow() > 1) cws.deleteRows(2, cws.getLastRow() - 1);
      const rev = bumpChatRevision();
      return out({ok:true, msg:"cleared", chatRev: rev});
    }

    if (data.action === "saveUsers") {
      const ws = getUserSheet();
      // Xóa hết dữ liệu cũ (giữ header)
      if (ws.getLastRow() > 1) ws.deleteRows(2, ws.getLastRow() - 1);
      const users = data.users || [];
      users.forEach(u => ws.appendRow([u.username, u.password, u.displayName || u.username, u.role || "viewer"]));
      return ok("saved");
    }

    if (data.action === "getUsers") {
      const ws = getUserSheet();
      const rows = ws.getDataRange().getValues();
      const users = rows.slice(1).filter(r => r[0]).map(r => ({
        username: r[0], password: r[1], displayName: r[2], role: r[3]
      }));
      return out({ok:true, users});
    }

  } catch(err) { return out({ok:false, error:err.toString()}); }
}

function doGet(e) {
  try {
    // Fallback gửi chat bằng GET (một số trình duyệt làm mất body POST khi redirect tới Google)
    if (e && e.parameter && e.parameter.action === "chat_send" && e.parameter.data) {
      try {
        const data = JSON.parse(e.parameter.data);
        if (data.action === "chat_send" && data.msg) {
          const ws = getChatSheet();
          pruneChatSheetOlderThanDays(ws);
          const m = data.msg;
          const cellText = chatCellTextFromMsg(m);
          ws.appendRow([m.id, m.username, m.displayName, cellText, m.ts]);
          return out({ ok: true, msg: "sent", savedText: cellText });
        }
      } catch (err2) { return out({ok:false, error:err2.toString()}); }
    }
    if (e && e.parameter && e.parameter.action === "chat_clear" && e.parameter.data) {
      try {
        const data = JSON.parse(e.parameter.data);
        if (data.action === "chat_clear") {
          const req = data.requestUser;
          if (!req) return out({ok:false, error:"no user"});
          const uws = getUserSheet();
          const urows = uws.getDataRange().getValues();
          const okAdmin = urows.slice(1).some(r => r[0] === req && String(r[3]).toLowerCase() === "admin");
          if (!okAdmin) return out({ok:false, error:"forbidden"});
          const cws = getChatSheet();
          if (cws.getLastRow() > 1) cws.deleteRows(2, cws.getLastRow() - 1);
          const rev = bumpChatRevision();
          return out({ok:true, msg:"cleared", chatRev: rev});
        }
      } catch (err3) { return out({ok:false, error:err3.toString()}); }
    }
    if (e && e.parameter && e.parameter.action === "chat_get") {
      const ws = getChatSheet();
      pruneChatSheetOlderThanDays(ws);
      const since = parseFloat(e.parameter.since || 0);
      var cutoff = Date.now() - CHAT_KEEP_DAYS * 24 * 60 * 60 * 1000;
      const rows = ws.getLastRow() > 1 ? ws.getRange(2,1,ws.getLastRow(),5).getValues() : [];
      const msgs = rows.filter(function(r) {
        var ts = parseFloat(r[4]);
        return r[0] && ts > since && ts >= cutoff;
      }).map(r => ({
        id: r[0], username: r[1], displayName: r[2], text: r[3], ts: parseFloat(r[4])
      }));
      return out({ok:true, msgs: msgs, chatRev: getChatRevision()});
    }
    if (e && e.parameter && e.parameter.action === "getUsers") {
      const ws = getUserSheet();
      const rows = ws.getDataRange().getValues();
      const users = rows.slice(1).filter(r => r[0]).map(r => ({
        username: r[0], password: r[1], displayName: r[2], role: r[3]
      }));
      return out({ok:true, users});
    }
    const ws = getSheet();
    const rows = ws.getDataRange().getValues();
    // Luôn trả về cả users trong mọi GET request
    const uws = getUserSheet();
    const urows = uws.getDataRange().getValues();
    const users = urows.slice(1).filter(r => r[0]).map(r => ({
      username: r[0], password: r[1], displayName: r[2], role: r[3]
    }));
    return out({ok:true, rows, users});
  } catch(err) { return out({ok:false, error:err.toString()}); }
}

function ok(msg) { return out({ok:true, msg}); }
function out(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
/** Chat: chỉ giữ ~15 ngày trên Sheet (tiết kiệm dung lượng / quota) */
var CHAT_KEEP_DAYS = 15;
function pruneChatSheetOlderThanDays(ws) {
  var cutoff = Date.now() - CHAT_KEEP_DAYS * 24 * 60 * 60 * 1000;
  var last = ws.getLastRow();
  if (last < 2) return;
  var data = ws.getRange(2, 1, last, 5).getValues();
  for (var i = data.length - 1; i >= 0; i--) {
    var ts = parseFloat(data[i][4]);
    if (data[i][0] && ts < cutoff) ws.deleteRow(i + 2);
  }
}
/** Nội dung cột text: upload ảnh/tệp lên Drive nếu có, không lưu base64 vào Sheet */
function chatCellTextFromMsg(m) {
  if (m.imageData) {
    var url = uploadImage(m.imageData, m.imageFileName || "chat.jpg");
    if (url) return "__CHAT_IMG__:" + url;
  }
  if (m.fileData && m.fileName) {
    var match = m.fileData.match(/^data:([^;]+);base64,(.+)$/);
    if (match) {
      try {
        var bytes = Utilities.base64Decode(match[2]);
        var blob = Utilities.newBlob(bytes, match[1], m.fileName);
        var folder = getImgFolder();
        var file = folder.createFile(blob);
        file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
        var u = "https://drive.google.com/uc?id=" + file.getId();
        return "__CHAT_FILE__:" + JSON.stringify({ url: u, name: m.fileName });
      } catch (e) {}
    }
  }
  return m.text || "";
}
function getChatSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let ws = findSheetByNameNormalized(ss, "Chat");
  if (!ws) {
    ws = ss.insertSheet("Chat");
    ws.appendRow(["id","username","displayName","text","ts"]);
  }
  return ws;
}
